#! /usr/bin/env bash 

sum()
{
    return $(($1+$2))
}

diff()
{
    return $(($2-$1))
}
div()
{
    return $(($2/$1))
}
mod()
{
    return $(($2 % $1))
}
mul()
{
    return $(($2 * $1))
}
exp()
{
    return $(($2 ** $1))
}

sum 10 20
echo "Sum is:"$?
diff 10 20
echo "Diff is:"$?
div 10 20
echo "Div of number is:"$?
mod 10 20
echo "Mod of number is:"$?
mul 10 20
echo "Mul of number is:"$?
exp 10 20
echo "Exp of number is:"$?
